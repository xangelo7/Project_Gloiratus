monthly_head_of_family_pulse =
{
	effect = {
		if = {
			limit = {
				is_general = no
				has_variable = disloyal_family_head_var
			}
			remove_variable = disloyal_family_head_var
			if = {
				limit = {
					has_triggered_character_modifier = disloyal_family_head_cmod
				}
				remove_triggered_character_modifier = disloyal_family_head_cmod
			}
		}
		if = {
			limit = {
				is_general = yes
				has_variable = disloyal_family_head_var
				NOT = {
					has_triggered_character_modifier = disloyal_family_head_cmod
				}
			}
			save_scope_as = revolt_target
			employer = {
				trigger_event = {
					id = family_revolt.2
				}
			}
		}
	}
	random_events =
	{
		100 = 0
		20 = family_revolt.1
	}
	events = {
		family_events.3
		family_events.4
	}
}
