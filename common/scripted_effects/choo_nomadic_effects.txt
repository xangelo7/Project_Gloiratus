set_proper_numof_chieftans = {


remove_country_modifier = chieftans_number_choom
every_in_list = {
variable = current_chieftans_list
employer = {
			add_country_modifier = {
				name = chieftans_number_choom
				duration = -1
				mode = add
			}
	}
}




}



generate_chieftan_candidates_list = {
	every_character = {
			limit = {
			is_male = yes
			is_adult = yes
			
			OR = {
			age >= 55
			is_general = yes
			finesse >= 8
			charisma >= 8
			}
			is_ruler = no 
			is_clan_chief = no
			
			PREV = {
			NOT = {
			is_target_in_variable_list = { name = nomination_for_chieftan_list target = PREV }
			is_target_in_variable_list = { name = current_chieftans_list target = PREV }
			}
			}

			}
			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = nomination_for_chieftan_list
					target = scope:to_add
				}
			}
		}
}


generate_chieftan_heir_candidates_list = {
	every_character = {
			limit = {
			is_clan_chief = yes
			NOR = {
			is_ruler = yes 
			}
			PREV = {
			NOT = {
			is_target_in_variable_list = { name = nomination_for_chieftan_heir_list target = PREV }
			}
			}

			}
			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = nomination_for_chieftan_heir_list
					target = scope:to_add
				}
			}
		}
}
