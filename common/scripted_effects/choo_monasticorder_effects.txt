
generate_devoted_candiates_list  = {

clear_variable_list = devoted_candiates_list

	every_character = {
	
			limit = {
			OR = {
			
			AND = {
			OR = {
			is_head_of_family = yes
			finesse >= 9
			charisma >= 9
			has_tech_office = yes
			is_primary_heir = yes
			is_child_of = root.current_ruler
			}
			employer = { government = seekers_councill } 
			}
			
			AND = {
			OR = {
			is_head_of_family = yes
			zeal >= 9
			is_general = yes
			is_primary_heir = yes
			is_child_of = root.current_ruler
			}
			employer = { government = monastic_order } 
			}

			}

			root = {
			NOT = {
			is_target_in_variable_list = { name = devoted_candiates_list target = PREV }
			}
			}
			}
			
			
			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = devoted_candiates_list
					target = scope:to_add
				}
			}
		
		}

		
}
