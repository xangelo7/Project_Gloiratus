
generate_leading_famillies_heads_list  = {

clear_variable_list = oligarchs_leading_famillies_heads

	every_character = {
	
			limit = {
			OR = {
			is_head_of_family = yes
			has_character_modifier = chosen_as_next_head_minister
			}
			root = {
			NOT = {
			is_target_in_variable_list = { name = oligarchs_leading_famillies_heads target = PREV }
			}
			}
			}
			
			
			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = oligarchs_leading_famillies_heads
					target = scope:to_add
				}
			}
		
		}

		
}

get_trade_posts_list = {

clear_variable_list = trade_posts_list

every_province = { 
limit = { has_variable = trade_post_constructed 

			NOT = {
			is_target_in_variable_list = { name = trade_posts_list target = this }
			}
}

			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = trade_posts_list
					target = scope:to_add
				}
			}

 }

}
