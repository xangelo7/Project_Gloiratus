add_country_modifier_stack = {
    if = {
        limit = {
            has_country_modifier = $MOD$
        }
        remove_country_modifier = $MOD$
    }
    while = {
        count = $NUM$
        add_country_modifier = {
            name = $MOD$
            mode = add
            duration = $DUR$
        }
    }
}
