

#race traits panel
show_racial_panel = {  
	
	scope = country
	
    is_shown = { 
	always = yes
    }
	ai_is_valid = {
	always = no
	}
	ai_chance = {
		base = 0
	}
	is_valid = {
    always = yes
    }
	effect = { 
	
	if = { limit = { has_variable = racial_panel_on } remove_variable = racial_panel_on } 
	else = { set_variable = racial_panel_on }
	
	}
}
racial_panel_is_shown = {
   
   scope = country
   	ai_is_valid = {
	 always = no
	}
	is_shown = { 
	has_variable = racial_panel_on
    }
	is_valid = {
    always = yes
    }

	effect = { 
	}
}
close_racial_panel = {

	scope = country
	ai_is_valid = {
	 always = no
	}
	ai_chance = {
		base = 10
	}
    is_shown = { 
	always = yes
    }
	is_valid = {
    always = yes
    }

	effect = { 
		remove_variable = racial_panel_on
	}
}
