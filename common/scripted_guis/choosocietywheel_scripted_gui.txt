#society wheel

show_wheel_of_society = {  
	
	scope = country
	
    is_shown = { 
	always = yes
    }
	is_valid = {
    always = no
    }
	effect = { 
	
	if = { limit = { has_variable = wheel_of_society_on  } remove_variable = wheel_of_society_on } 
	else = { set_variable = wheel_of_society_on  }
	
	}
}
	
wheel_of_society_is_shown = {

	scope = country

	is_shown = { 

	has_variable = wheel_of_society_on

	}
	is_valid = {
    always = yes
    }

	effect = {
	
	}
}

wheel_of_society_might_1 = {

	scope = country

	is_shown = { 
	
	always = yes

	}
	
	ai_is_valid = { 
	always = no
	}
	
	is_valid = {
    has_variable = society_points
	var:society_points >= society_development_level_cost
	NOT = { has_variable = fearless_wariors  }
    }

	effect = {
	set_variable = fearless_wariors
	change_variable = { name = society_points subtract = society_development_level_cost }
	change_variable = { name = society_development_level add = 0.5 } 
	
	set_variable = mightful_society_way
	
	}
}
