military_colonies = {

	duration = 0
	enable = no
	toggle = no
	army_only = yes
	
	map = yes
	soundeffect ="event:/SFX/Animations/Units/Generic/sfx_anim_unit_build_colony"
	
	
	hidden = {
		unit_owner = { 
			NOR = { 
				has_military_tradition = greek_philosophy
				has_military_tradition = levantine_philosophy
				has_military_tradition = persian_philosophy 
			} 
		}
	}

	allow = {
		unit_owner = { 
			trigger_if = {
				limit = { has_military_tradition = greek_philosophy }
				has_military_bonus = greek_states_path_4
			}
			trigger_else_if = {
				limit = { has_military_tradition = levantine_philosophy }
				has_military_bonus = levantine_egyptian_path_4
			}
			trigger_else_if = {
				limit = { has_military_tradition = persian_philosophy }
				has_military_bonus = persian_bactrian_path_4
			}
		}
		has_commander = yes
		is_army = yes
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		unit_size >= 5
		custom_tooltip = {
			text = "num_of_loyal_cohorts_tooltip"
			num_of_non_loyal_cohorts_svalue >= 1
		}
		hidden:unit_owner = { can_pay_price = create_military_colony }
		hidden:unit_location = {
			has_owner = yes
			total_population <= 5
			state_loyalty >= 33
			NOT = {
				state = {
					any_state_province = {
						is_capital = yes
					}
				}
			}
			owner = root.unit_owner 
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	
	start_effect = {
		unit_owner = {  pay_price = create_military_colony }
		if = {
			limit = {
				exists = commander
			}
			random_sub_unit = {
				limit = {
					has_personal_loyalty = no
				}
				if = {
					limit = {
						root = {
							unit_owner = {
								is_republic = yes
							}
						}
					}
					set_personal_loyalty = root.commander
				}
				else_if = {
					limit = {
						root = {
							commander = {
								is_ruler = no
							}
						} 
					}
					set_personal_loyalty = root.commander
				}
				else = {
					root = {
						unit_owner = {
							add_tyranny = 1
						}
					}
				}
			}
		}
		unit_location = {
			create_state_pop = freemen
			add_state_loyalty = -15
		}
		
	}
	
	
}
