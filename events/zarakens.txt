zaraken.3 = {
	type = country_event
	title = zaraken.3.t
	desc = zaraken.3.desc
	picture = greek_city_market
	
	left_portrait = current_ruler

	trigger = {}

	immediate = {
		p:4588 = {
			save_scope_as = jarenam_capital_scope
		}
	}

	option = {      
		name = zaraken.3.a #Staying at home
		hidden_effect = {
			scope:jarenam_capital_scope = {
				create_country = {
					name = {
						name = "JARENAM_NAME"
						adjective = "JARENAM_ADJECTIVE"
					}
					save_scope_as = jarenam_scope
					change_country_tag = JR9
					change_country_flag = JR9_FLAG 
					change_country_color = navy_blue
					set_primary_culture = zaraken_1
					set_country_religion = zainuddian
					change_government = aristocratic_monarchy
					add_treasury = 50
				}
			}
			every_owned_province = {
				limit = {
					OR = {
						is_in_area = shout_eteradica_area_1
						is_in_area = shout_eteradica_area_2
						is_in_area = shout_eteradica_area_4
						is_in_area = shout_eteradica_area_3
					}
				}
				set_owned_by = scope:jarenam_scope
			}
			if = {
				limit = {
					NOT = { is_subject_of = root }
				}
				make_subject = { # Make Colony
					target = scope:jarenam_scope
					type = princedom_vassal
				}
			}
		}
	}
}
