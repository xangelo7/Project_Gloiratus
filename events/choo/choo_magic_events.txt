namespace = choo_magic
#magic influx growth
choo_magic.1 = {
	type = country_event
	hidden = yes
	
	trigger = {
		any_owned_province = { num_of_tmages >= 1 }
	}
	immediate = { 	
		if = { 
			limit = { 
				NOT = { has_variable = magic_influx } 
			} 
			set_variable = { name = magic_influx value = 0 } 
		}
		change_variable = { name = magic_influx add = magic_influx_growth_value } 	
	}	
}
