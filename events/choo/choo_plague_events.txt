namespace = choo_plague
#plague spread mechanics
choo_plague.1 = {
	type = country_event
	hidden = yes
	
	trigger = {
		any_owned_province = { 
		has_variable = has_plague
		}	
	}
	immediate = {  	
		every_country_state = { 	
			limit = { 
				any_state_province = { has_variable = has_plague } 
			}
			random_state_province = { 
				limit = { 
					NOT = {  has_variable = has_plague }  
				} 
				random = { chance = plague_chance_of_spread 
				spread_the_plague = yes
				} 
			}
		}	
	}	
}

#plague end
choo_plague.2 = {
	type = province_event
	hidden = yes	
	trigger = {
		has_variable = has_plague
	}
	
	immediate = {  	
		remove_variable = has_plague	
		change_devastation_level_by = { AMOUNT = -5 }	
	}	
}
