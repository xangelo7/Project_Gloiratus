namespace = comet

comet.1 = {
	type = country_event
	title = "comet.1.t"
	desc = "comet.1.desc"
	
	left_portrait = current_ruler
	right_portrait = scope:priest
	
	picture = natural_disaster
	
	trigger = {
		has_land = yes
		any_neighbour_country = {
			has_land = yes
		}
		NOT = {
			has_variable = comet_sighted
		}
		any_character = {
			OR = {
				has_office = office_high_priest_monarchy
				has_office = office_augur
				has_office = office_pontifex
				has_office = office_high_priest
			}
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		random_neighbour_country = {
			limit = {
				has_land = yes
			} 
			random_owned_province = {
				save_scope_as = target
			}
		}
		set_variable = {
			name = comet_sighted
			value = scope:target
		}
		random_character = {
			limit = {
				OR = {
					has_office = office_high_priest_monarchy
					has_office = office_augur
					has_office = office_pontifex
					has_office = office_high_priest
				}
			}
			save_scope_as = priest
		}
		every_neighbour_country = {
			limit = {
				has_land = yes
				any_neighbour_country = {
					has_land = yes
				}
				NOT = {
					has_variable = comet_sighted
				}
				any_character = {
					OR = {
						has_office = office_high_priest_monarchy
						has_office = office_augur
						has_office = office_pontifex
						has_office = office_high_priest
					}
				}
			}
			trigger_event = {
				id = comet.1
				days = 2
			}
		}
	}
	
	option = { 
		name = "comet.1.a"
		add_stability = 10
		custom_tooltip = "comet_sighted_1_tt"
	}
	 
}

comet.4 = {
	type = country_event
	title = "comet.4.t"
	desc = "comet.4.desc"
	
	picture = city_country
	
	trigger = {
		has_country_modifier = comet_sighted_modifier
		has_variable = comet_sighted
		var:comet_sighted = {
			has_owner = yes
			owner = root
			is_capital = no
		}
		war = yes
		manpower >= 10
		NOT = {
			has_variable = comet_moved_toggle
		}
	}
	
	immediate = {
		var:comet_sighted = {
			save_scope_as = target
		}
		set_variable = {
			name = comet_moved_toggle
		}
	}
	
	option = {
		name = "comet.4.a"
		add_manpower = -10
		add_war_exhaustion = 4
		add_stability = 20
	}
	
	option = {
		name = "comet.4.b"
		remove_country_modifier = comet_sighted_modifier
	}
	
}


comet.6 = {
	type = country_event
	title = "comet.6.t"
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:demodamas_cl
					root = scope:demodamas_cl
				}
				desc = "comet.6.repeat_desc"
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = "comet.6.regular_desc"
			}
		}
	}
	picture = ship_in_storm
	
	left_portrait = current_ruler
	right_portrait = scope:demodamas
	
	trigger = {
		has_land = yes
		exists = scope:demodamas
		scope:demodamas = {
			is_alive = yes
			employer = c:REB
		}
	}
	
	option = {
		name = "comet.6.a"
		create_country_treasure = {
			key = "candlestick_of_great_delights"
			icon = "treasure_chest"
			
			modifier = {
				happiness_for_wrong_culture_group_modifier = 0.1
			}
		}
		add_treasury = {
			value = 0
			subtract = has_monthly_income
			multiply = 6
		}
		scope:demodamas = {
			if = {
				limit = {
					NOT = {
						has_trait = ambitious
					}
				}
				add_trait = ambitious
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = scholar
					}
				}
				add_trait = scholar
			}
		}
		custom_tooltip = "demodamas_goes_away"
		hidden_effect = {
			save_scope_as = demodamas_cl2
			random_country = {
				limit = {
					is_ai = no
				}
				weight = {
					modifier = {
						factor = 0.001
						AND = {
							exists = scope:demodamas_cl2
							this = scope:demodamas_cl2
						}
					}
					modifier = {
						factor = 0.001
						AND = {
							exists = scope:demodamas_cl
							this = scope:demodamas_cl
						} 
					}
				}
				trigger_event = {
					id = comet.7
					days = { 730 1450 }
					#days = 1
				}
			}
		}
	}
	
	option = {
		name = "comet.6.b"
		current_ruler = {
			add_popularity = subtract_popularity_large
		}
		scope:demodamas = {
			if = {
				limit = {
					NOT = {
						has_trait = lunatic
					}
				}
				add_trait = lunatic
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = brain_damaged
					}
				}
				add_trait = brain_damaged
			}
		}
		custom_tooltip = "demodamas_goes_away"
		hidden_effect = {
			save_scope_as = demodamas_cl2
			random_country = {
				limit = {
					is_ai = no
				}
				weight = {
					modifier = {
						factor = 0.001
						AND = {
							exists = scope:demodamas_cl2
							this = scope:demodamas_cl2
						}
					}
					modifier = {
						factor = 0.001
						AND = {
							exists = scope:demodamas_cl
							this = scope:demodamas_cl
						} 
					}
				}
				trigger_event = {
					id = comet.7
					days = { 730 1450 }
				}
			}
		}
	}
	
	option = {
		name = "comet.6.c"
		scope:demodamas = {
			move_country = root
			set_as_minor_character = THIS
		}
		if = {
			limit = {
				exists = scope:demodamas_cl
				scope:demodamas_cl = {
					is_ai = no
					NOT = {
						this = root
					}
				}
			}
			save_scope_as = took_in_demodamas
			scope:demodamas_cl = {
				trigger_event = {
					id = comet.8
					days = 1 #Correct
				}
			}
		}
		custom_tooltip = "demodamas_settles_down"
	}
	
}
